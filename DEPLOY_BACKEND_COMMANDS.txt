# Copy and paste these commands ONE AT A TIME into your terminal

# 1. Go to backend folder
cd "/Users/cr/Desktop/HYT Game/backend"

# 2. Login to Railway (this will open your browser)
railway login

# 3. Create Railway project
railway init

# 4. Set environment variables
railway variables set JWT_SECRET="hyt-jwt-secret-$(openssl rand -hex 16)"
railway variables set PORT=3000
railway variables set FRONTEND_URL="https://hyt-game.vercel.app"

# 5. Deploy backend
railway up

# 6. Get your backend URL (copy the URL it shows)
railway status

# ===== AFTER YOU GET THE BACKEND URL =====

# 7. Go back to project root
cd "/Users/cr/Desktop/HYT Game"

# 8. Add backend URL to Vercel (it will ask you to type it in)
vercel env add VITE_API_URL production
# When it asks for the value, type: https://YOUR-RAILWAY-URL/api
# (replace YOUR-RAILWAY-URL with the URL from step 6)

# 9. Redeploy frontend
vercel --prod

# DONE! Go to https://hyt-game.vercel.app and it will work!
